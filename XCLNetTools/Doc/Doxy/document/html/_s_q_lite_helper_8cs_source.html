413  public static DataTable SelectPaging(string connString, string tableName, string strColumns, string strWhere, string strOrder, int pageSize, int currentIndex, out int recordOut)
414  {
415  DataTable dt = new DataTable();
416  recordOut = Convert.ToInt32(ExecuteScalar(connString, "select count(*) from " + tableName, CommandType.Text));
417  string pagingTemplate = "select {0} from {1} where {2} order by {3} limit {4} offset {5} ";
418  int offsetCount = (currentIndex - 1) * pageSize;
419  string commandText = String.Format(pagingTemplate, strColumns, tableName, strWhere, strOrder, pageSize.ToString(), offsetCount.ToString());
420  using (DbDataReader reader = ExecuteReader(connString, commandText, CommandType.Text))
421  {
422  if (reader != null)
423  {
424  dt.Load(reader);
425  }
426  }
427  return dt;
428  }

440  private static void PrepareCommand(SQLiteCommand cmd, SQLiteConnection conn, ref SQLiteTransaction trans, bool useTrans, CommandType cmdType, string cmdText, params SQLiteParameter[] cmdParms)
441  {
442  if (conn.State != ConnectionState.Open)
443  conn.Open();
444 
445  cmd.Connection = conn;
446  cmd.CommandText = cmdText;
447 
448  if (useTrans)
449  {
450  trans = conn.BeginTransaction(IsolationLevel.ReadCommitted);
451  cmd.Transaction = trans;
452  }
453 
454  cmd.CommandType = cmdType;
455 
456  if (cmdParms != null)
457  {
458  foreach (SQLiteParameter parm in cmdParms)
459  cmd.Parameters.Add(parm);
460  }
461  }
462  }
