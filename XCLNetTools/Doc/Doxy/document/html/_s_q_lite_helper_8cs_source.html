445  public static DataTable SelectPaging(string connString, string tableName, string strColumns, string strWhere, string strOrder, int pageSize, int currentIndex, out int recordOut)
446  {
447  DataTable dt = new DataTable();
448  recordOut = Convert.ToInt32(ExecuteScalar(connString, "select count(*) from " + tableName, CommandType.Text));
449  string pagingTemplate = "select {0} from {1} where {2} order by {3} limit {4} offset {5} ";
450  int offsetCount = (currentIndex - 1) * pageSize;
451  string commandText = String.Format(pagingTemplate, strColumns, tableName, strWhere, strOrder, pageSize.ToString(), offsetCount.ToString());
452  using (DbDataReader reader = ExecuteReader(connString, commandText, CommandType.Text))
453  {
454  if (reader != null)
455  {
456  dt.Load(reader);
457  }
458  }
459  return dt;
460  }

472  private static void PrepareCommand(SQLiteCommand cmd, SQLiteConnection conn, ref SQLiteTransaction trans, bool useTrans, CommandType cmdType, string cmdText, params SQLiteParameter[] cmdParms)
473  {
474  if (conn.State != ConnectionState.Open)
475  conn.Open();
476 
477  cmd.Connection = conn;
478  cmd.CommandText = cmdText;
479 
480  if (useTrans)
481  {
482  trans = conn.BeginTransaction(IsolationLevel.ReadCommitted);
483  cmd.Transaction = trans;
484  }
485 
486  cmd.CommandType = cmdType;
487 
488  if (cmdParms != null)
489  {
490  foreach (SQLiteParameter parm in cmdParms)
491  cmd.Parameters.Add(parm);
492  }
493  }
494  }
