29  {

100  public static IList<T> DataTableToList<T>(DataTable dt) where T : new()
101  {
102  if (null == dt || null == dt.Rows || dt.Rows.Count == 0)
103  {
104  return new List<T>();
105  }
106  IList<T> ts = new List<T>();
107  foreach (DataRow dr in dt.Rows)
108  {
109  T t = new T();
110  PropertyInfo[] propertys = t.GetType().GetProperties();
111  foreach (PropertyInfo pi in propertys)
112  {
113  if (!dt.Columns.Contains(pi.Name) || !pi.CanWrite)
114  {
115  continue;
116  }
117  var value = dr[pi.Name];
118  if (value != DBNull.Value)
119  {
120  pi.SetValue(t, value, null);
121  }
122  }
123  ts.Add(t);
124  }
125  return ts;
126  }
