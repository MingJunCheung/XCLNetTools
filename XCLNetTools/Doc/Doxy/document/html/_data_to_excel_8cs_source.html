68  #region 合法性检测

70  if (null == paramClass.Ds || paramClass.Ds.Tables.Count == 0)

73  }
74  if (null != paramClass.ConTitle && null != paramClass.Ds)
75  {
76  if (paramClass.ConTitle.Length != paramClass.Ds.Tables.Count)

83  }
84  }
85  if (null != paramClass.OutPutClass && paramClass.OutPutClass.Count > 0)
86  {
87  if (paramClass.TableName.Length != paramClass.Ds.Tables.Count)
88  {
89  str.Append("表名与dataSet的table数量不一致，导出失败！；");
90  }
91  }
92  if (str.Length > 0)
93  {

98

100

103  {
104  workbook = new Workbook(paramClass.WorkBookFilePath);
105  }

109  }
110 
111  #endregion 是否指定被操作的工作薄
112

114  {

116

118  {
119  sheet.Name = paramClass.ConTitle[i];

121

123  {
124  workbook.Worksheets.Add();
125  }
126

128  List<string> dtColNameLst = new List<string>();

132  }
133

135 
136  List<string> newNamesLst = new List<string>();
137  if (null != paramClass.OutPutClass && paramClass.OutPutClass.Count > 0)

140  foreach (var m in outPutModel.Fields)
141  {
142  newNamesLst.Add(m.newName);
143  if (dtColNameLst.Contains(m.oldName))
144  {
145  dt.Columns[m.oldName].ColumnName = m.newName;
146  }
147  }
148  }

153 
154  #endregion 写入列名
155

157 
158  dt.AcceptChanges();
159  sheet.Cells.ImportDataTable(dt.DefaultView.ToTable("dtNew", true, newNamesLst.ToArray()), paramClass.IsShowFieldLine, paramClass.FirstRowIndex, paramClass.FirstColumnIndex, dt.Rows.Count + 1, dt.Columns.Count, true, "yyyy-MM-dd HH:mm:ss");
160  sheet.AutoFitColumns();
161 
162  #endregion 向sheet中写入数据
163

166  Cells cells = sheet.Cells;

169  cells[0, 0].Value = string.Format("数据导出：{0}；导出时间：{1}；记录总数：{2}", paramClass.ConTitle[i], DateTime.Now, dt.Rows.Count);
170  Aspose.Cells.Style styleCell0 = cells[0, 0].GetStyle();
171  styleCell0.Font.Color = System.Drawing.Color.Red;
172  cells[0, 0].SetStyle(styleCell0);
173  }

176  Range range = sheet.Cells.CreateRange(1, 0, 1, newNamesLst.Count);
177  range.Name = "Range1";

192 
193  if (!string.IsNullOrEmpty(paramClass.CustomFileName))
194  {

196  }

200  bool isFirefox = HttpContext.Current.Request.Browser.Type.ToLower().Contains("firefox");
