29  {
30  List<XCLNetTools.Entity.KeyValue> cookielist = new List<XCLNetTools.Entity.KeyValue>();
31  foreach (string item in cookie.Split(new string[] { ";", "," }, StringSplitOptions.RemoveEmptyEntries))
32  {
33  if (Regex.IsMatch(item, @"([\s\S]*?)=([\s\S]*?)$"))
34  {
35  Match m = Regex.Match(item, @"([\s\S]*?)=([\s\S]*?)$");
36  cookielist.Add(new XCLNetTools.Entity.KeyValue() { Key = m.Groups[1].Value, Value = m.Groups[2].Value });
37  }
38  }
