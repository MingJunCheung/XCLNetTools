61  public HttpResult GetHtml(HttpItem item)
62  {
63  //返回参数
64  HttpResult result = new HttpResult();
65  try
66  {
67  //准备参数
68  SetRequest(item);
69  }
70  catch (Exception ex)
71  {
72  result = new HttpResult();
73  result.Cookie = string.Empty;
74  result.Header = null;
75  result.Html = ex.Message;
76  result.StatusDescription = "配置参数时出错：" + ex.Message;
77  return result;
78  }
79  try
80  {
81  #region 得到请求的response
82 
83  using (response = (HttpWebResponse)request.GetResponse())
84  {
85  result.StatusCode = response.StatusCode;
86  result.StatusDescription = response.StatusDescription;
87  result.Header = response.Headers;
88  if (response.Cookies != null) result.CookieCollection = response.Cookies;
89  if (response.Headers["set-cookie"] != null) result.Cookie = response.Headers["set-cookie"];
90  MemoryStream _stream = new MemoryStream();
91  //GZIIP处理
92  if (response.ContentEncoding != null && response.ContentEncoding.Equals("gzip", StringComparison.InvariantCultureIgnoreCase))
93  {
94  //开始读取流并设置编码方式
95  //new GZipStream(response.GetResponseStream(), CompressionMode.Decompress).CopyTo(_stream, 10240);
96  //.net4.0以下写法
97  _stream = GetMemoryStream(new GZipStream(response.GetResponseStream(), CompressionMode.Decompress));
98  }
99  else
100  {
101  //开始读取流并设置编码方式
102  //response.GetResponseStream().CopyTo(_stream, 10240);
103  //.net4.0以下写法
104  _stream = GetMemoryStream(response.GetResponseStream());
105  }
106  //获取Byte
107  byte[] ResponseByte = _stream.ToArray();
108  _stream.Close();
109  if (ResponseByte != null & ResponseByte.Length > 0)
110  {
111  //是否返回Byte类型数据
112  if (item.ResultType == ResultType.Byte) result.ResultByte = ResponseByte;
113  //从这里开始我们要无视编码了
114  if (encoding == null)
115  {
116  Match meta = Regex.Match(Encoding.Default.GetString(ResponseByte), "<meta([^<]*)charset=([^<]*)[\"']", RegexOptions.IgnoreCase);
117  string c = (meta.Groups.Count > 1) ? meta.Groups[2].Value.ToLower() : string.Empty;
118  if (c.Length > 2)
119  encoding = Encoding.GetEncoding(c.Trim().Replace("\"", "").Replace("'", "").Replace(";", "").Replace("iso-8859-1", "gbk"));
120  else
121  {
122  if (string.IsNullOrEmpty(response.CharacterSet)) encoding = Encoding.UTF8;
123  else encoding = Encoding.GetEncoding(response.CharacterSet);
124  }
125  }
126  //得到返回的HTML
127  result.Html = encoding.GetString(ResponseByte);
128  }
129  else
130  {
131  //得到返回的HTML
132  result.Html = "本次请求并未返回任何数据";
133  }
134  }
135 
136  #endregion 得到请求的response
137  }
138  catch (WebException ex)
139  {
140  //这里是在发生异常时返回的错误信息
141  response = (HttpWebResponse)ex.Response;
142  result.Html = ex.Message;
143  if (response != null)
144  {
145  result.StatusCode = response.StatusCode;
146  result.StatusDescription = response.StatusDescription;
147  }
148  }
149  catch (Exception ex)
150  {
151  result.Html = ex.Message;
152  }
153  if (item.IsToLower) result.Html = result.Html.ToLower();
154  return result;
155  }

277  private void SetPostData(HttpItem item)
278  {
279  //验证在得到结果时是否有传入数据
280  if (request.Method.Trim().ToLower().Contains("post"))
281  {
282  if (item.PostEncoding != null)
283  {
284  postencoding = item.PostEncoding;
285  }
286  byte[] buffer = null;
287  //写入Byte类型
288  if (item.PostDataType == PostDataType.Byte && item.PostdataByte != null && item.PostdataByte.Length > 0)
289  {
290  //验证在得到结果时是否有传入数据
291  buffer = item.PostdataByte;
292  }//写入文件
293  else if (item.PostDataType == PostDataType.FilePath && !string.IsNullOrEmpty(item.Postdata))
294  {
295  StreamReader r = new StreamReader(item.Postdata, postencoding);
296  buffer = postencoding.GetBytes(r.ReadToEnd());
297  r.Close();
298  } //写入字符串
299  else if (!string.IsNullOrEmpty(item.Postdata))
300  {
301  buffer = postencoding.GetBytes(item.Postdata);
302  }
303  if (buffer != null)
304  {
305  request.ContentLength = buffer.Length;
306  request.GetRequestStream().Write(buffer, 0, buffer.Length);
307  }
308  }
