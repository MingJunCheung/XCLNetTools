48  public static string GetClientIP()
49  {
50  var context = HttpContext.Current;
51  if (null == context)
52  {
53  return string.Empty;
54  }
55  string result = string.Empty;
56  try
57  {
58  string ipAddress = string.Empty;
59  if (context.Request.ServerVariables.HasKeys() && context.Request.ServerVariables.AllKeys.ToList().Exists(k => string.Equals(k, "HTTP_X_FORWARDED_FOR", StringComparison.OrdinalIgnoreCase)))
60  {
61  ipAddress = context.Request.ServerVariables["HTTP_X_FORWARDED_FOR"];
62  if (!string.IsNullOrEmpty(ipAddress))
63  {
64  string[] addresses = ipAddress.Split(',');
65  if (addresses.Length != 0)
66  {
67  result = addresses[0];
68  }
69  }
70  }
71 
72  if (string.IsNullOrEmpty(result))
73  {
74  result = context.Request.ServerVariables["REMOTE_ADDR"];
75  }
76  }
77  catch
78  {
79  }
80  return string.Equals("::1", result) ? "127.0.0.1" : result;
81  }

187  public static string NoHTML(string html)
188  {
189  if (string.IsNullOrEmpty(html))
190  {
191  return "";
192  }
193  //删除脚本
194  html = Regex.Replace(html, @"<script[^>]*?>.*?</script>", "", RegexOptions.IgnoreCase);
195  //删除HTML
196  html = Regex.Replace(html, @"<(.[^>]*)>", "", RegexOptions.IgnoreCase);
197  html = Regex.Replace(html, @"([\r\n])[\s]+", "", RegexOptions.IgnoreCase);
198  html = Regex.Replace(html, @"-->", "", RegexOptions.IgnoreCase);
199  html = Regex.Replace(html, @"<!--.*", "", RegexOptions.IgnoreCase);
200  html = Regex.Replace(html, @"&(quot|#34);", "\"", RegexOptions.IgnoreCase);
201  html = Regex.Replace(html, @"&(amp|#38);", "&", RegexOptions.IgnoreCase);
202  html = Regex.Replace(html, @"&(lt|#60);", "<", RegexOptions.IgnoreCase);
203  html = Regex.Replace(html, @"&(gt|#62);", ">", RegexOptions.IgnoreCase);
204  html = Regex.Replace(html, @"&(nbsp|#160);", " ", RegexOptions.IgnoreCase);
205  html = Regex.Replace(html, @"&(iexcl|#161);", "\xa1", RegexOptions.IgnoreCase);
206  html = Regex.Replace(html, @"&(cent|#162);", "\xa2", RegexOptions.IgnoreCase);
207  html = Regex.Replace(html, @"&(pound|#163);", "\xa3", RegexOptions.IgnoreCase);
208  html = Regex.Replace(html, @"&(copy|#169);", "\xa9", RegexOptions.IgnoreCase);
209  html = Regex.Replace(html, @"&#(\d+);", "", RegexOptions.IgnoreCase);
210  html = html.Replace("\r\n", "");
211  html = html.Replace("'", "");
212  html = HttpContext.Current.Server.HtmlEncode(html).Trim();
213  return html;

398  public static string GZipCompressString(string text)
399  {
400  if (string.IsNullOrEmpty(text))
401  {
402  return string.Empty;
403  }
404  byte[] buffer = Encoding.UTF8.GetBytes(text);
405 
406  using (var memoryStream = new MemoryStream())
407  {
408  using (var gZipStream = new GZipStream(memoryStream, CompressionMode.Compress, true))
409  {
410  gZipStream.Write(buffer, 0, buffer.Length);
411  }
412  memoryStream.Position = 0;
413  var compressedData = new byte[memoryStream.Length];
414  memoryStream.Read(compressedData, 0, compressedData.Length);
415 
416  var gZipBuffer = new byte[compressedData.Length + 4];
417  Buffer.BlockCopy(compressedData, 0, gZipBuffer, 4, compressedData.Length);
418  Buffer.BlockCopy(BitConverter.GetBytes(buffer.Length), 0, gZipBuffer, 0, 4);
419  return Convert.ToBase64String(gZipBuffer);
420  }
421  }

428  public static string GZipDecompressString(string compressedText)
429  {
430  if (string.IsNullOrEmpty(compressedText))
431  {
432  return string.Empty;
433  }
434  byte[] gZipBuffer = Convert.FromBase64String(compressedText);
435  using (var memoryStream = new MemoryStream())
436  {
437  int dataLength = BitConverter.ToInt32(gZipBuffer, 0);
438  memoryStream.Write(gZipBuffer, 4, gZipBuffer.Length - 4);
439  var buffer = new byte[dataLength];
440  memoryStream.Position = 0;
441  using (var gZipStream = new GZipStream(memoryStream, CompressionMode.Decompress))
442  {
443  gZipStream.Read(buffer, 0, buffer.Length);
444  }
445  return Encoding.UTF8.GetString(buffer);
446  }
