98  public static string NoHTML(string html)
99  {
100  if (string.IsNullOrEmpty(html))
101  {
102  return "";
103  }
104  //删除脚本
105  html = Regex.Replace(html, @"<script[^>]*?>.*?</script>", "", RegexOptions.IgnoreCase);
106  //删除HTML
107  html = Regex.Replace(html, @"<(.[^>]*)>", "", RegexOptions.IgnoreCase);
108  html = Regex.Replace(html, @"([\r\n])[\s]+", "", RegexOptions.IgnoreCase);
109  html = Regex.Replace(html, @"-->", "", RegexOptions.IgnoreCase);
110  html = Regex.Replace(html, @"<!--.*", "", RegexOptions.IgnoreCase);
111  html = Regex.Replace(html, @"&(quot|#34);", "\"", RegexOptions.IgnoreCase);
112  html = Regex.Replace(html, @"&(amp|#38);", "&", RegexOptions.IgnoreCase);
113  html = Regex.Replace(html, @"&(lt|#60);", "<", RegexOptions.IgnoreCase);
114  html = Regex.Replace(html, @"&(gt|#62);", ">", RegexOptions.IgnoreCase);
115  html = Regex.Replace(html, @"&(nbsp|#160);", " ", RegexOptions.IgnoreCase);
116  html = Regex.Replace(html, @"&(iexcl|#161);", "\xa1", RegexOptions.IgnoreCase);
117  html = Regex.Replace(html, @"&(cent|#162);", "\xa2", RegexOptions.IgnoreCase);
118  html = Regex.Replace(html, @"&(pound|#163);", "\xa3", RegexOptions.IgnoreCase);
119  html = Regex.Replace(html, @"&(copy|#169);", "\xa9", RegexOptions.IgnoreCase);
120  html = Regex.Replace(html, @"&#(\d+);", "", RegexOptions.IgnoreCase);
121  html = html.Replace("\r\n", "");
122  html = html.Replace("'", "");
123  html = HttpContext.Current.Server.HtmlEncode(html).Trim();
124  return html;
125  }
