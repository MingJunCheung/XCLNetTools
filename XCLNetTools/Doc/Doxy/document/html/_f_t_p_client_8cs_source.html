792  private Socket CreateDataSocket()
793  {
794  SendCommand("PASV");
795  if (iReplyCode != 227)
796  {
797  throw new IOException(strReply.Substring(4));
798  }
799  int index1 = strReply.IndexOf('(');
800  int index2 = strReply.IndexOf(')');
801  string ipData = strReply.Substring(index1 + 1, index2 - index1 - 1);
802  int[] parts = new int[6];
803  int len = ipData.Length;
804  int partCount = 0;
805  string buf = "";
806  for (int i = 0; i < len && partCount <= 6; i++)
807  {
808  char ch = Char.Parse(ipData.Substring(i, 1));
809  if (Char.IsDigit(ch))
810  buf += ch;
811  else if (ch != ',')
812  {
813  throw new IOException("Malformed PASV strReply: " + strReply);
814  }
815  if (ch == ',' || i + 1 == len)
816  {
817  try
818  {
819  parts[partCount++] = Int32.Parse(buf);
820  buf = "";
821  }
822  catch (Exception)
823  {
824  throw new IOException("Malformed PASV strReply: " + strReply);
825  }
826  }
827  }
828  string ipAddress = parts[0] + "." + parts[1] + "." + parts[2] + "." + parts[3];
829  int port = (parts[4] << 8) + parts[5];
830  Socket s = new Socket(AddressFamily.InterNetwork, SocketType.Stream, ProtocolType.Tcp);
831  IPEndPoint ep = new IPEndPoint(IPAddress.Parse(ipAddress), port);
832  try
833  {
834  s.Connect(ep);
835  }
836  catch (Exception)
837  {
838  throw new IOException("无法连接ftp服务器");
839  }
840  return s;
841  }
