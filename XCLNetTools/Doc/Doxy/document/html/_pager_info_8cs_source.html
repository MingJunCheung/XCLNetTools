43  private void Refresh()
44  {
45  if (this.RecordCount > 0)
46  {
47  this.PageCount = (int)Math.Ceiling((1.0 * this.RecordCount) / this.PageSize);
48  this._pageIndex = this.PageIndex > this.PageCount ? this.PageCount : this.PageIndex;
49  this.CurrentPageRecordCount = this.PageIndex == this.PageCount && this.RecordCount % this.PageSize > 0 ? this.RecordCount % this.PageSize : this.PageSize;
50  this.StartIndex = (this.PageIndex - 1) * this.PageSize + 1;
51  this.EndIndex = this.StartIndex + this.CurrentPageRecordCount - 1;
52  }
53  else
54  {
55  this.PageCount = 0;
56  this.CurrentPageRecordCount = 0;
57  this.StartIndex = 0;
