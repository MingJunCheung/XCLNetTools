46  private void Refresh()
47  {
48  if (this.RecordCount > 0)
49  {
50  this.PageCount = (int)Math.Ceiling((1.0 * this.RecordCount) / this.PageSize);
51  this._pageIndex = this.PageIndex > this.PageCount ? this.PageCount : this.PageIndex;
52  this.CurrentPageRecordCount = this.PageIndex == this.PageCount && this.RecordCount % this.PageSize > 0 ? this.RecordCount % this.PageSize : this.PageSize;
53  this.StartIndex = (this.PageIndex - 1) * this.PageSize + 1;
54  this.EndIndex = this.StartIndex + this.CurrentPageRecordCount - 1;
