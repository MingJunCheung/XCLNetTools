26  public static IList<T> DataReaderToIList<T>(IDataReader dr) where T : new()
27  {
28  if (null == dr)
29  {
30  return null;
31  }
32  IList<T> lst = new List<T>();
33  var fields = new List<string>();
34  using (dr)
35  {
36  for (var i = 0; i < dr.FieldCount; i++)
37  {
38  fields.Add(dr.GetName(i));
39  }
40  while (dr.Read())
41  {
42  var t = new T();
43  PropertyInfo[] propertys = t.GetType().GetProperties();
44  foreach (PropertyInfo pi in propertys)
45  {
46  if (!fields.Contains(pi.Name) || !pi.CanWrite)
47  {
48  continue;
49  }
50  var value = dr[pi.Name];
51  if (value != DBNull.Value)
52  {
53  pi.SetValue(t, value, null);
54  }
55  }
56  lst.Add(t);
57  }
58  }
59  return lst;
60  }
