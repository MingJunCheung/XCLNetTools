77  private static void AssignParameterValues(SqlParameter[] commandParameters, DataRow dataRow)
78  {
79  if ((commandParameters == null) || (dataRow == null))
80  {
81  return;
82  }
83 
84  int i = 0;
85  // 设置参数值
86  foreach (SqlParameter commandParameter in commandParameters)
87  {
88  // 创建参数名称,如果不存在,只抛出一个异常.
89  if (commandParameter.ParameterName == null ||
90  commandParameter.ParameterName.Length <= 1)
91  throw new Exception(
92  string.Format("请提供参数{0}一个有效的名称{1}.", i, commandParameter.ParameterName));
93  // 从dataRow的表中获取为参数数组中数组名称的列的索引.
94  // 如果存在和参数名称相同的列,则将列值赋给当前名称的参数.
95  if (dataRow.Table.Columns.IndexOf(commandParameter.ParameterName.Substring(1)) != -1)
96  commandParameter.Value = dataRow[commandParameter.ParameterName.Substring(1)];
97  i++;
98  }

754 
755  if (connectionOwnership == SqlConnectionOwnership.External)
