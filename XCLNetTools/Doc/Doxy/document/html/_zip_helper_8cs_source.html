28  public static void ToZipFile(string fileToZip, string zipedFile, int compressionLevel, int blockSize)
29  {
30  //如果文件没有找到，则报错
31  if (!System.IO.File.Exists(fileToZip))
32  {
33  throw new System.IO.FileNotFoundException("指定要压缩的文件: " + fileToZip + " 不存在!");
34  }
35  using (System.IO.FileStream ZipFile = System.IO.File.Create(zipedFile))
36  {
37  using (ZipOutputStream ZipStream = new ZipOutputStream(ZipFile))
38  {
39  using (System.IO.FileStream StreamToZip = new System.IO.FileStream(fileToZip, System.IO.FileMode.Open, System.IO.FileAccess.Read))
40  {
41  string fileName = fileToZip.Substring(fileToZip.LastIndexOf("\\") + 1);
42  ZipEntry ZipEntry = new ZipEntry(fileName);
43  ZipStream.PutNextEntry(ZipEntry);
44  ZipStream.SetLevel(compressionLevel);
45  byte[] buffer = new byte[blockSize];
46  int sizeRead = 0;
47  try
48  {
49  do
50  {
51  sizeRead = StreamToZip.Read(buffer, 0, buffer.Length);
52  ZipStream.Write(buffer, 0, sizeRead);
53  }
54  while (sizeRead > 0);
55  }
56  catch
57  {
58  throw;
59  }
60  StreamToZip.Close();
61  }
62  ZipStream.Finish();
63  ZipStream.Close();
64  }
65  ZipFile.Close();
66  }

124  private static void ZipSetp(string strDirectory, ZipOutputStream s, string parentPath)
125  {
126  if (strDirectory[strDirectory.Length - 1] != Path.DirectorySeparatorChar)
127  {
128  strDirectory += Path.DirectorySeparatorChar;
129  }
130  Crc32 crc = new Crc32();
131  string[] filenames = Directory.GetFileSystemEntries(strDirectory);
132  foreach (string file in filenames)// 遍历所有的文件和目录
133  {
134  if (Directory.Exists(file))// 先当作目录处理如果存在这个目录就递归Copy该目录下面的文件
135  {
136  string pPath = parentPath;
137  pPath += file.Substring(file.LastIndexOf("\\") + 1);
138  pPath += "\\";
139  ZipSetp(file, s, pPath);
140  }
141  else // 否则直接压缩文件
142  {
143  //打开压缩文件
144  using (FileStream fs = File.OpenRead(file))
145  {
146  byte[] buffer = new byte[fs.Length];
147  fs.Read(buffer, 0, buffer.Length);
148  string fileName = parentPath + file.Substring(file.LastIndexOf("\\") + 1);
149  ZipEntry entry = new ZipEntry(fileName);
150  entry.DateTime = DateTime.Now;
151  entry.Size = fs.Length;
152  fs.Close();
153  crc.Reset();
154  crc.Update(buffer);
155  entry.Crc = crc.Value;
156  s.PutNextEntry(entry);
157  s.Write(buffer, 0, buffer.Length);
158  }
159  }
160  }

170  public static void UnZip(string zipedFile, string strDirectory, string password, bool overWrite)
171  {
172  if (strDirectory == "")
173  strDirectory = Directory.GetCurrentDirectory();
174  if (!strDirectory.EndsWith("\\"))
175  strDirectory = strDirectory + "\\";
176  using (ZipInputStream s = new ZipInputStream(File.OpenRead(zipedFile)))
177  {
178  s.Password = password;
179  ZipEntry theEntry;
180  while ((theEntry = s.GetNextEntry()) != null)
181  {
182  string directoryName = "";
183  string pathToZip = "";
184  pathToZip = theEntry.Name;
185  if (pathToZip != "")
186  directoryName = Path.GetDirectoryName(pathToZip) + "\\";
187  string fileName = Path.GetFileName(pathToZip);
188  Directory.CreateDirectory(strDirectory + directoryName);
189  if (fileName != "")
190  {
191  if ((File.Exists(strDirectory + directoryName + fileName) && overWrite) || (!File.Exists(strDirectory + directoryName + fileName)))
192  {
193  using (FileStream streamWriter = File.Create(strDirectory + directoryName + fileName))
194  {
195  int size;
196  byte[] data = new byte[2048];
197  while (true)
198  {
199  size = s.Read(data, 0, data.Length);
200  if (size > 0)
201  streamWriter.Write(data, 0, size);
202  else
203  break;
204  }
205  streamWriter.Close();
206  }
207  }
208  }

220  public static void CompressFile(string[] files, string zipedFile, params string[] fileNames)
221  {
222  bool isCustomFileName = false;
223  if (null != fileNames && fileNames.Length > 0)
224  {
225  if (files.Length != fileNames.Length)
226  {
227  throw new ArgumentException("指定了自定义的文件名的数目必须与文件数一致！");
228  }
229  else
230  {
231  isCustomFileName = true;
232  }
233  }
234  if (null != files && files.Length > 0)
235  {
236  using (FileStream ZipFile = File.Create(zipedFile))
237  {
238  using (ZipOutputStream ZipStream = new ZipOutputStream(ZipFile))
239  {
240  for (int i = 0; i < files.Length; i++)
241  {
242  string fileToZip = files[i];
243  if (!System.IO.File.Exists(fileToZip))
244  {
245  continue;
246  }
247  using (FileStream fs = File.OpenRead(fileToZip))
248  {
249  byte[] buffer = new byte[fs.Length];
250  fs.Read(buffer, 0, buffer.Length);
251  fs.Close();
252  string fileName = isCustomFileName ? fileNames[i] : fileToZip.Substring(fileToZip.LastIndexOf("\\") + 1);
253  ZipEntry ZipEntry = new ZipEntry(fileName);
254  ZipStream.PutNextEntry(ZipEntry);
255  ZipStream.SetLevel(5);
256  ZipStream.Write(buffer, 0, buffer.Length);
257  }
258  }
259  ZipStream.Finish();
260  ZipStream.Close();
261  }
262  }
263  }
