61  public static string AddParam(string url, NameValueCollection param)
62  {
63  if (string.IsNullOrEmpty(url) || (null == param || param.Count == 0))
64  {
65  return url;
66  }
67 
68  url = url.Trim().TrimEnd(new char[] { '?', '&', ' ' }).Trim();
69 
70  Uri uri = null;
71  if (!Uri.TryCreate(url, UriKind.RelativeOrAbsolute, out uri))
72  {
73  return url;
74  }
75 
76  string firstStr = string.Empty;
77  if (uri.IsAbsoluteUri)
78  {
79  //绝对
80  firstStr = string.IsNullOrEmpty(uri.Query) ? (url + "?") : (url + "&");
81  }
82  else
83  {
84  //相对
85  firstStr = url.Contains("?") ? (url + "&") : (url + "?");
86  }
87 
88  List<string> paramLst = new List<string>();
89  foreach (string m in param)
90  {
91  paramLst.Add(string.Format("{0}={1}", m, HttpContext.Current.Server.UrlEncode(param[m])));
92  }
93 
94  return firstStr + string.Join("&", paramLst.ToArray());
95  }

104  public static string UpdateParam(string url, string paramName, string value)
105  {

108  int index1 = url.IndexOf("&", index);

126  public static string RemoveParam(string url, string paramName)
127  {
128  string[] s = url.Split('?');
129  if (s.Length == 2 && s[1].Length > 0)
130  {
131  string[] ps = s[1].Split('&');
132  if (ps.Length > 0)
133  {
134  Dictionary<string, string> lst = new Dictionary<string, string>();
135  for (int i = 0; i < ps.Length; i++)
136  {
137  string[] keys = ps[i].Split('=');
138  if (keys.Length == 2)
139  {
140  if (!lst.ContainsKey(keys[0]))
141  {
142  lst.Add(keys[0], keys[1]);
143  }
144  }
145  }
146  lst.Remove(paramName);
147  if (null != lst && lst.Count > 0)
148  {
149  List<string> urlLst = new List<string>();
150  foreach (KeyValuePair<string, string> entry in lst)
151  {
152  urlLst.Add(string.Format("{0}={1}", entry.Key, entry.Value));
153  }
154  url = string.Format("{0}?{1}", s[0], string.Join("&", urlLst.ToArray()));
155  }
156  else
157  {
158  url = s[0];
159  }
160  }
161  }
162  return url;
163  }

175  public static void ParseUrl(string url, out string baseUrl, out NameValueCollection nvc)
176  {
177  if (url == null)
178  throw new ArgumentNullException("url");
179 
180  nvc = new NameValueCollection();
181  baseUrl = "";
182 
183  if (url == "")
184  return;
185 
186  int questionMarkIndex = url.IndexOf('?');
187 
188  if (questionMarkIndex == -1)
189  {
190  baseUrl = url;
191  return;
192  }
193  baseUrl = url.Substring(0, questionMarkIndex);
194  if (questionMarkIndex == url.Length - 1)
195  return;
196  string ps = url.Substring(questionMarkIndex + 1);
197 
198  // 开始分析参数对
199  Regex re = new Regex(@"(^|&)?(\w+)=([^&]+)(&|$)?", RegexOptions.Compiled);
200  MatchCollection mc = re.Matches(ps);
201 
202  foreach (Match m in mc)
203  {
204  nvc.Add(m.Result("$2").ToLower(), m.Result("$3"));
205  }
206  }

238  public static int GetUrlError(string curl)
239  {
240  int num = 200;
241  HttpWebRequest request = (HttpWebRequest)WebRequest.Create(new Uri(curl));
242  ServicePointManager.Expect100Continue = false;
243  try

245  ((HttpWebResponse)request.GetResponse()).Close();
246  }
247  catch (WebException exception)

254  {
255  return 500;
