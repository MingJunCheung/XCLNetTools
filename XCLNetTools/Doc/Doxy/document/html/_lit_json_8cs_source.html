39  public static string ConvertDataTableToArray(System.Data.DataTable dt)
40  {
41  string str = string.Empty;
42  if (null != dt && dt.Rows.Count > 0)
43  {
44  List<string> strTemp = new List<string>();
45  string columnName, dataType, value;//列名，字段类型，值
46  for (int i = 0; i < dt.Rows.Count; i++)
47  {
48  StringBuilder strBuild = new StringBuilder();
49  for (int j = 0; j < dt.Columns.Count; j++)
50  {
51  columnName = dt.Columns[j].ColumnName;
52  dataType = dt.Columns[j].DataType.Name;
53  value = Convert.ToString(dt.Rows[i][j]);
54  JsonWriter jw = new JsonWriter();
55  jw.WriteObjectStart();
56  jw.WritePropertyName(columnName);
57  WriteFormat(jw, value, dataType);
58  jw.WriteObjectEnd();
59  strBuild.Append(jw.ToString());
60  }
61  strTemp.Add(strBuild.ToString().Replace("}{", ","));
62  }
63  str = String.Join(",", strTemp.ToArray());
64  }
65  return string.Format("[{0}]", str);
66  }

157  private static void WriteFormat(JsonWriter jw, string value, string dataType)
158  {
159  switch (dataType)
160  {
161  case "Int32":
162  Int32 int32Value;
163  Int32.TryParse(value, out int32Value);
164  jw.Write(int32Value);
165  break;
166 
167  case "Boolean":
168  Boolean booleanValue;
169  Boolean.TryParse(value, out booleanValue);
170  jw.Write(booleanValue);
171  break;
172 
173  case "Decimal":
174  Decimal decimalValue;
175  Decimal.TryParse(value, out decimalValue);
176  jw.Write(decimalValue);
177  break;
178 
179  case "Double":
180  Double doubleValue;
181  Double.TryParse(value, out doubleValue);
182  jw.Write(doubleValue);
183  break;
184 
185  default:
186  jw.Write(value);
187  break;
188  }
