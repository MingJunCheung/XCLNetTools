250  private byte[] GetData(HttpWebRequest request)
251  {
252  HttpWebResponse response = (HttpWebResponse)request.GetResponse();
253  Stream stream = response.GetResponseStream();
254  responseHeaders = response.Headers;
255 
256  DownloadEventArgs args = new DownloadEventArgs();
257  if (responseHeaders[HttpResponseHeader.ContentLength] != null)
258  args.TotalBytes = Convert.ToInt32(responseHeaders[HttpResponseHeader.ContentLength]);
259 
260  MemoryStream ms = new MemoryStream();
261  int count = 0;
262  byte[] buf = new byte[bufferSize];
263  while ((count = stream.Read(buf, 0, buf.Length)) > 0)
264  {
265  ms.Write(buf, 0, count);
266  if (this.DownloadProgressChanged != null)
267  {
268  args.BytesReceived += count;
269  args.ReceivedData = new byte[count];
270  Array.Copy(buf, args.ReceivedData, count);
271  this.DownloadProgressChanged(this, args);
272  }
273  }
274  stream.Close();
275  //解压
276  if (ResponseHeaders[HttpResponseHeader.ContentEncoding] != null)
277  {
278  MemoryStream msTemp = new MemoryStream();
279  count = 0;
280  buf = new byte[100];
281  switch (ResponseHeaders[HttpResponseHeader.ContentEncoding].ToLower())
282  {
283  case "gzip":
284  GZipStream gzip = new GZipStream(ms, CompressionMode.Decompress);
285  while ((count = gzip.Read(buf, 0, buf.Length)) > 0)
286  {
287  msTemp.Write(buf, 0, count);
288  }
289  return msTemp.ToArray();
290 
291  case "deflate":
292  DeflateStream deflate = new DeflateStream(ms, CompressionMode.Decompress);
293  while ((count = deflate.Read(buf, 0, buf.Length)) > 0)
294  {
295  msTemp.Write(buf, 0, count);
296  }
297  return msTemp.ToArray();
298 
299  default:
300  break;
301  }
302  }
303  return ms.ToArray();
304  }

321  if (sendBufferSize > remainBytes) sendBufferSize = remainBytes;

344  ServicePointManager.ServerCertificateValidationCallback = new RemoteCertificateValidationCallback(this.CheckValidationResult);

348  HttpWebRequest.DefaultCachePolicy = policy;

350  HttpWebRequest request = (HttpWebRequest)WebRequest.Create(uri);
351  request.AllowAutoRedirect = false;
352  request.AllowWriteStreamBuffering = false;
353  request.Method = method;
354  if (proxy != null)

356  request.CookieContainer = cc;
