277  private byte[] GetData(HttpWebRequest request)
278  {
279  HttpWebResponse response = (HttpWebResponse)request.GetResponse();
280  Stream stream = response.GetResponseStream();
281  responseHeaders = response.Headers;
282  //SaveCookiesToDisk();
283 
284  DownloadEventArgs args = new DownloadEventArgs();
285  if (responseHeaders[HttpResponseHeader.ContentLength] != null)
286  args.TotalBytes = Convert.ToInt32(responseHeaders[HttpResponseHeader.ContentLength]);
287 
288  MemoryStream ms = new MemoryStream();
289  int count = 0;
290  byte[] buf = new byte[bufferSize];
291  while ((count = stream.Read(buf, 0, buf.Length)) > 0)
292  {
293  ms.Write(buf, 0, count);
294  if (this.DownloadProgressChanged != null)
295  {
296  args.BytesReceived += count;
297  args.ReceivedData = new byte[count];
298  Array.Copy(buf, args.ReceivedData, count);
299  this.DownloadProgressChanged(this, args);
300  }
301  }
302  stream.Close();
303  //解压
304  if (ResponseHeaders[HttpResponseHeader.ContentEncoding] != null)
305  {
306  MemoryStream msTemp = new MemoryStream();
307  count = 0;
308  buf = new byte[100];
309  switch (ResponseHeaders[HttpResponseHeader.ContentEncoding].ToLower())
310  {
311  case "gzip":
312  GZipStream gzip = new GZipStream(ms, CompressionMode.Decompress);
313  while ((count = gzip.Read(buf, 0, buf.Length)) > 0)
314  {
315  msTemp.Write(buf, 0, count);
316  }
317  return msTemp.ToArray();
318 
319  case "deflate":
320  DeflateStream deflate = new DeflateStream(ms, CompressionMode.Decompress);
321  while ((count = deflate.Read(buf, 0, buf.Length)) > 0)
322  {
323  msTemp.Write(buf, 0, count);
324  }
325  return msTemp.ToArray();
326 
327  default:
328  break;
329  }
330  }
331  return ms.ToArray();
332  }

339  private void PostData(HttpWebRequest request, byte[] postData)
340  {
341  int offset = 0;
342  int sendBufferSize = bufferSize;
343  int remainBytes = 0;
344  Stream stream = request.GetRequestStream();
345  UploadEventArgs args = new UploadEventArgs();
346  args.TotalBytes = postData.Length;
347  while ((remainBytes = postData.Length - offset) > 0)
348  {
349  if (sendBufferSize > remainBytes) sendBufferSize = remainBytes;
350  stream.Write(postData, offset, sendBufferSize);
351  offset += sendBufferSize;
352  if (this.UploadProgressChanged != null)
353  {
354  args.BytesSent = offset;
355  this.UploadProgressChanged(this, args);
356  }
357  }
358  stream.Close();
359  }

367  private HttpWebRequest CreateRequest(string url, string method)
368  {
369  Uri uri = new Uri(url);
370 
371  if (uri.Scheme == "https")
372  ServicePointManager.ServerCertificateValidationCallback = new RemoteCertificateValidationCallback(this.CheckValidationResult);
373 
374  // Set a default policy level for the "http:" and "https" schemes.
375  HttpRequestCachePolicy policy = new HttpRequestCachePolicy(HttpRequestCacheLevel.Revalidate);
376  HttpWebRequest.DefaultCachePolicy = policy;
377 
378  HttpWebRequest request = (HttpWebRequest)WebRequest.Create(uri);
379  request.AllowAutoRedirect = false;
380  request.AllowWriteStreamBuffering = false;
381  request.Method = method;
382  if (proxy != null)
383  request.Proxy = proxy;
384  request.CookieContainer = cc;
385  foreach (string key in requestHeaders.Keys)
386  {
387  request.Headers.Add(key, requestHeaders[key]);
388  }
389  requestHeaders.Clear();
