293  private byte[] GetData(HttpWebRequest request)
294  {
295  HttpWebResponse response = (HttpWebResponse)request.GetResponse();
296  Stream stream = response.GetResponseStream();
297  responseHeaders = response.Headers;
298  //SaveCookiesToDisk();
299 
300  DownloadEventArgs args = new DownloadEventArgs();
301  if (responseHeaders[HttpResponseHeader.ContentLength] != null)
302  args.TotalBytes = Convert.ToInt32(responseHeaders[HttpResponseHeader.ContentLength]);
303 
304  MemoryStream ms = new MemoryStream();
305  int count = 0;
306  byte[] buf = new byte[bufferSize];
307  while ((count = stream.Read(buf, 0, buf.Length)) > 0)
308  {
309  ms.Write(buf, 0, count);
310  if (this.DownloadProgressChanged != null)
311  {
312  args.BytesReceived += count;
313  args.ReceivedData = new byte[count];
314  Array.Copy(buf, args.ReceivedData, count);
315  this.DownloadProgressChanged(this, args);
316  }
317  }
318  stream.Close();
319  //解压
320  if (ResponseHeaders[HttpResponseHeader.ContentEncoding] != null)
321  {
322  MemoryStream msTemp = new MemoryStream();
323  count = 0;
324  buf = new byte[100];
325  switch (ResponseHeaders[HttpResponseHeader.ContentEncoding].ToLower())
326  {
327  case "gzip":
328  GZipStream gzip = new GZipStream(ms, CompressionMode.Decompress);
329  while ((count = gzip.Read(buf, 0, buf.Length)) > 0)
330  {
331  msTemp.Write(buf, 0, count);
332  }
333  return msTemp.ToArray();
334 
335  case "deflate":
336  DeflateStream deflate = new DeflateStream(ms, CompressionMode.Decompress);
337  while ((count = deflate.Read(buf, 0, buf.Length)) > 0)
338  {
339  msTemp.Write(buf, 0, count);
340  }
341  return msTemp.ToArray();
342 
343  default:
344  break;
345  }
346  }
347  return ms.ToArray();
348  }

355  private void PostData(HttpWebRequest request, byte[] postData)
356  {
357  int offset = 0;
358  int sendBufferSize = bufferSize;
359  int remainBytes = 0;
360  Stream stream = request.GetRequestStream();
361  UploadEventArgs args = new UploadEventArgs();
362  args.TotalBytes = postData.Length;
363  while ((remainBytes = postData.Length - offset) > 0)
364  {
365  if (sendBufferSize > remainBytes) sendBufferSize = remainBytes;
366  stream.Write(postData, offset, sendBufferSize);
367  offset += sendBufferSize;
368  if (this.UploadProgressChanged != null)
369  {
370  args.BytesSent = offset;
371  this.UploadProgressChanged(this, args);
372  }
373  }
374  stream.Close();
375  }

383  private HttpWebRequest CreateRequest(string url, string method)
384  {
385  Uri uri = new Uri(url);
386 
387  if (uri.Scheme == "https")
388  ServicePointManager.ServerCertificateValidationCallback = new RemoteCertificateValidationCallback(this.CheckValidationResult);
389 
390  // Set a default policy level for the "http:" and "https" schemes.
391  HttpRequestCachePolicy policy = new HttpRequestCachePolicy(HttpRequestCacheLevel.Revalidate);
392  HttpWebRequest.DefaultCachePolicy = policy;
393 
394  HttpWebRequest request = (HttpWebRequest)WebRequest.Create(uri);
395  request.AllowAutoRedirect = false;
396  request.AllowWriteStreamBuffering = false;
397  request.Method = method;
398  if (proxy != null)
399  request.Proxy = proxy;
400  request.CookieContainer = cc;
401  foreach (string key in requestHeaders.Keys)
402  {
403  request.Headers.Add(key, requestHeaders[key]);
404  }
405  requestHeaders.Clear();
406  return request;
