39  private static SqlParameter[] DiscoverSpParameterSet(SqlConnection connection, string spName, bool includeReturnValueParameter)
40  {
41  if (connection == null) throw new ArgumentNullException("connection");
42  if (spName == null || spName.Length == 0) throw new ArgumentNullException("spName");
43 
44  SqlCommand cmd = new SqlCommand(spName, connection);
45  cmd.CommandType = CommandType.StoredProcedure;
46 
47  connection.Open();
48  // 检索cmd指定的存储过程的参数信息,并填充到cmd的Parameters参数集中.
49  SqlCommandBuilder.DeriveParameters(cmd);
50  connection.Close();
51  // 如果不包含返回值参数,将参数集中的每一个参数删除.
52  if (!includeReturnValueParameter)
53  {
54  cmd.Parameters.RemoveAt(0);
55  }
56 
57  // 创建参数数组
58  SqlParameter[] discoveredParameters = new SqlParameter[cmd.Parameters.Count];
59  // 将cmd的Parameters参数集复制到discoveredParameters数组.
60  cmd.Parameters.CopyTo(discoveredParameters, 0);
61 
62  // 初始化参数值为 DBNull.Value.
63  foreach (SqlParameter discoveredParameter in discoveredParameters)
64  {
65  discoveredParameter.Value = DBNull.Value;
66  }
67  return discoveredParameters;
68  }

212  private static SqlParameter[] GetSpParameterSetInternal(SqlConnection connection, string spName, bool includeReturnValueParameter)
213  {
214  if (connection == null) throw new ArgumentNullException("connection");
215  if (spName == null || spName.Length == 0) throw new ArgumentNullException("spName");
216 
217  string hashKey = connection.ConnectionString + ":" + spName + (includeReturnValueParameter ? ":include ReturnValue Parameter" : "");
218 
219  SqlParameter[] cachedParameters;
220 
221  cachedParameters = paramCache[hashKey] as SqlParameter[];
222  if (cachedParameters == null)
223  {
224  SqlParameter[] spParameters = DiscoverSpParameterSet(connection, spName, includeReturnValueParameter);
225  paramCache[hashKey] = spParameters;
226  cachedParameters = spParameters;
227  }
228 
229  return CloneParameters(cachedParameters);
