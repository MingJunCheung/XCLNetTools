189  public static decimal GetTimeSub(DateTime? start, DateTime? end, string workStart, string workEnd)
190  {
191  decimal sum = 0m;
192  DateTime dtNow = DateTime.Now;
193  DateTime dtStart = XCLNetTools.Common.DataTypeConvert.ToDateTime(Convert.ToString(start), dtNow);
194  DateTime dtEnd = XCLNetTools.Common.DataTypeConvert.ToDateTime(Convert.ToString(end), dtNow);
195  if (dtStart > dtEnd)
196  {
197  return 0;
198  }
199  DateTime oldStartDay = Convert.ToDateTime(dtStart.ToShortDateString());
200  DateTime oldEndDay = Convert.ToDateTime(dtEnd.ToShortDateString());
201  String oldStartStr = dtStart.ToShortTimeString();
202  String oldEndStr = dtEnd.ToShortTimeString();
203  String newStartStr = workStart;//如："08:00" 工作时间（始）
204  String newEndStr = workEnd;//如："18:00" 工作时间（终）
205  String[] oldStart = oldStartStr.Split(':');
206  String[] oldEnd = oldEndStr.Split(':');
207  String[] newStart = newStartStr.Split(':');
208  String[] newEnd = newEndStr.Split(':');
209  int oldStartSec = XCLNetTools.Common.DataTypeConvert.ToInt(oldStart[0]) * 3600 + XCLNetTools.Common.DataTypeConvert.ToInt(oldStart[1]) * 60;
210  int oldEndSec = XCLNetTools.Common.DataTypeConvert.ToInt(oldEnd[0]) * 3600 + XCLNetTools.Common.DataTypeConvert.ToInt(oldEnd[1]) * 60;
211  int newStartSec = XCLNetTools.Common.DataTypeConvert.ToInt(newStart[0]) * 3600 + XCLNetTools.Common.DataTypeConvert.ToInt(newStart[1]) * 60;
212  int newEndSec = XCLNetTools.Common.DataTypeConvert.ToInt(newEnd[0]) * 3600 + XCLNetTools.Common.DataTypeConvert.ToInt(newEnd[1]) * 60;
213  if (String.Equals(oldStartDay, oldEndDay))
214  { //开始日期和结束日期是同一天
215  String strStart = (oldStartSec > newStartSec) ? oldStartStr : newStartStr;
216  String strEnd = (oldEndSec > newEndSec) ? newEndStr : oldEndStr;
217  DateTime dtSamDayStartTime = XCLNetTools.Common.DataTypeConvert.ToDateTime(String.Format("{0} {1}", String.Format("{0:yyyy-MM-dd}", oldStartDay), strStart));
218  DateTime dtSamDayEndTime = XCLNetTools.Common.DataTypeConvert.ToDateTime(String.Format("{0} {1}", String.Format("{0:yyyy-MM-dd}", oldEndDay), strEnd));
219  if (dtSamDayStartTime <= dtSamDayEndTime)
220  {
221  sum = Convert.ToDecimal(Math.Round(dtSamDayEndTime.Subtract(dtSamDayStartTime).TotalSeconds / 3600.0, 1));
222  }
223  }
224  else
225  { //开始日期和结束日期不是同一天，且结束日期大于开始日期（数据验证中有验证）
226  int span = oldEndDay.Subtract(oldStartDay).Days;
227  DateTime dt = oldStartDay;
228  String strStart = (oldStartSec > newStartSec) ? oldStartStr : newStartStr;
229  String strEnd = (oldEndSec > newEndSec) ? newEndStr : oldEndStr;
230  String st = String.Empty;
231  String se = String.Empty;
232  for (int i = 0; i <= span; i++)
233  {
234  st = i == 0 ? strStart : newStartStr;
235  se = i == span ? strEnd : newEndStr;
236  DateTime dtDiffStartTime = XCLNetTools.Common.DataTypeConvert.ToDateTime(String.Format("{0} {1}", String.Format("{0:yyyy-MM-dd}", dt), st));
237  DateTime dtDiffEndTime = XCLNetTools.Common.DataTypeConvert.ToDateTime(String.Format("{0} {1}", String.Format("{0:yyyy-MM-dd}", dt), se));
238  if (dtDiffStartTime <= dtDiffEndTime)
239  {
240  sum += Convert.ToDecimal(Math.Round(dtDiffEndTime.Subtract(dtDiffStartTime).TotalSeconds / (3600.0), 1));
241  }
242  }
243  }
244  return sum;
245  }

415  private static string Rep_date(Match mc)
416  {
417  string cnd = XCLNetTools.Common.Consts.CNDigit;
418  string val = mc.Value;
419  string digit = val.Substring(0, val.Length - 1);
420  char c = val[val.Length - 1];
421  val = "";
422  switch (c)
423  {
424  case '年':
425  foreach (char s in digit) val += cnd[int.Parse(s.ToString())];
426  break;
427 
428  case '月':
429  case '日':
430  if (digit.Length == 1)
431  val += cnd[int.Parse(digit)];
432  else
433  {
434  val += cnd[int.Parse(digit[0].ToString())] + "十";
435  val = val.TrimStart('一');
436  val += cnd[int.Parse(digit[1].ToString())].ToString().Trim('〇');
437  }
438  break;
439 
440  default:
441  return "格式错误";
442  }
443  return val + c.ToString();
444  }
