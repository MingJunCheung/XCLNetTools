97  public static int GetOrSetBlockFromHex(int blockIndex, int source, int? newBlockValue = null)
98  {
99  if (blockIndex <= 0)
100  {
101  throw new ArgumentException("blockIndex必须大于0");
102  }
103  if (newBlockValue.HasValue && (newBlockValue < 0 || newBlockValue > 15))
104  {
105  throw new ArgumentException("blockValue必须>=0并且<=15");
106  }
107 
108  int moveBits = (blockIndex - 1) * 4;
109  int blockValue = (source & (0xF << moveBits)) >> moveBits;
110 
111  if (newBlockValue.HasValue)
112  {
113  return source - (blockValue << moveBits) + (newBlockValue.Value << moveBits);
114  }
115  else
116  {
117  return blockValue;
