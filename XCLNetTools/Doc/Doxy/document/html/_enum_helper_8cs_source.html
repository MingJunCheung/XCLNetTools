60  }
61  var tp = typeof(T);
62  object objVal = null;

64  System.Reflection.FieldInfo[] fields = emType.GetFields();

68  {
69  var model = new XCLNetTools.Entity.Enum.EnumFieldTModel<T>();
70  objVal = emType.InvokeMember(field.Name, BindingFlags.GetField, null, null, null);
71  model.Value = (T)Convert.ChangeType(objVal, typeof(T));

77  }
78  list.Add(model);
