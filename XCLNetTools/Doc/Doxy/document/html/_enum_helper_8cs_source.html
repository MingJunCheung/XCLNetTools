340  for (int j = 0; j < fields.Count; j++)
341  {
342  string val = fields[j].Name;
343  string des = "";
344 
345  Object[] customObjs = fields[j].GetCustomAttributes(typeof(DescriptionAttribute), false);
346  if (null != customObjs && customObjs.Length > 0)

349  }
350 
351  str.AppendFormat(@"""{0}"":""{1}""", val, des);
352  if (j != fields.Count - 1)
353  {

356  }
