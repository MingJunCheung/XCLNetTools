134  public static void DownLoadFile(string path, string realName)
135  {
136  path = ComFile.MapPath(path);
137  realName = string.IsNullOrEmpty(realName) ? "文件下载" : realName;
138  FileInfo info = new FileInfo(path);
139  HttpResponse Response = HttpContext.Current.Response;
140  Response.Clear();
141  Response.ClearHeaders();
142  Response.Buffer = false;
143  Response.ContentType = "application/octet-stream";
144  string exp = HttpContext.Current.Request.UserAgent.ToLower();
145  if (exp.IndexOf("msie") > -1)
146  {
147  //当客户端使用IE时，对其进行编码；We should encode the filename when our visitors use IE
148  //使用 ToHexString 代替传统的 UrlEncode()；We use "ToHexString" replaced "context.Server.UrlEncode(fileName)"
149  realName = XCLNetTools.Encode.Hex.ToHexString(realName);
150  }
151  Response.AppendHeader("Content-Disposition", "attachment;filename=" + realName.Replace("+", "_").Replace(" ", ""));
152  Response.AppendHeader("Content-Length", info.Length.ToString());
153  Response.WriteFile(path);
154  Response.Flush();
155  Response.End();
156  }
