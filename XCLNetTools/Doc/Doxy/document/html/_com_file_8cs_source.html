150  public static void DownLoadFile(string path, string realName)
151  {
152  path = ComFile.MapPath(path);
153  realName = string.IsNullOrEmpty(realName) ? "文件下载" : realName;
154  FileInfo info = new FileInfo(path);
155  HttpResponse Response = HttpContext.Current.Response;
156  Response.Clear();
157  Response.ClearHeaders();
158  Response.Buffer = false;
159  Response.ContentType = "application/octet-stream";
160  string exp = HttpContext.Current.Request.UserAgent.ToLower();
161  if (exp.IndexOf("msie") > -1)
162  {
163  //当客户端使用IE时，对其进行编码；We should encode the filename when our visitors use IE
164  //使用 ToHexString 代替传统的 UrlEncode()；We use "ToHexString" replaced "context.Server.UrlEncode(fileName)"
165  realName = XCLNetTools.Encode.Hex.ToHexString(realName);
166  }
167  Response.AppendHeader("Content-Disposition", "attachment;filename=" + realName.Replace("+", "_").Replace(" ", ""));
168  Response.AppendHeader("Content-Length", info.Length.ToString());
169  Response.WriteFile(path);
170  Response.Flush();
171  Response.End();
172  }
