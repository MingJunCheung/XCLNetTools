61  public static void MakeThumbnail(string originalImagePath, string thumbnailPath, int width, int height, XCLNetTools.Enum.CommonEnum.ThumbImageModeEnum mode = XCLNetTools.Enum.CommonEnum.ThumbImageModeEnum.EqualRatioWH)
62  {
63  System.Drawing.Image originalImage = System.Drawing.Image.FromFile(originalImagePath);
64  if (null == originalImage)
65  {
66  return;
67  }
68  if (width <= 0 || height <= 0)
69  {
70  throw new ArgumentException("width or height is invalid!");
71  }
72 
73  int towidth = width;
74  int toheight = height;
75 
76  int x = 0;
77  int y = 0;
78  int ow = originalImage.Width;
79  int oh = originalImage.Height;
80 
81  switch (mode)
82  {
83  case XCLNetTools.Enum.CommonEnum.ThumbImageModeEnum.W:
84  toheight = originalImage.Height * width / originalImage.Width;
85  break;
86 
87  case XCLNetTools.Enum.CommonEnum.ThumbImageModeEnum.H:
88  towidth = originalImage.Width * height / originalImage.Height;
89  break;
90 
91  case XCLNetTools.Enum.CommonEnum.ThumbImageModeEnum.EqualRatioWH:
92  if ((double)originalImage.Width / (double)originalImage.Height > (double)towidth / (double)toheight)
93  {
94  oh = originalImage.Height;
95  ow = originalImage.Height * towidth / toheight;
96  y = 0;
97  x = (originalImage.Width - ow) / 2;
98  }
99  else
100  {
101  ow = originalImage.Width;
102  oh = originalImage.Width * height / towidth;
103  x = 0;
104  y = (originalImage.Height - oh) / 2;
105  }
106  break;
107 
108  case XCLNetTools.Enum.CommonEnum.ThumbImageModeEnum.WH:
109  default:
110  break;
111  }
112  //新建一个bmp图片
113  System.Drawing.Image bitmap = new System.Drawing.Bitmap(towidth, toheight);
114  //新建一个画板
115  System.Drawing.Graphics g = System.Drawing.Graphics.FromImage(bitmap);
116  //设置高质量插值法
117  g.InterpolationMode = System.Drawing.Drawing2D.InterpolationMode.High;
118  //设置高质量,低速度呈现平滑程度
119  g.SmoothingMode = System.Drawing.Drawing2D.SmoothingMode.HighQuality;
120  //清空画布并以透明背景色填充
121  g.Clear(System.Drawing.Color.Transparent);
122  //在指定位置并且按指定大小绘制原图片的指定部分
123  g.DrawImage(originalImage, new System.Drawing.Rectangle(0, 0, towidth, toheight), new System.Drawing.Rectangle(x, y, ow, oh), System.Drawing.GraphicsUnit.Pixel);
124  try
125  {
126  //以jpg格式保存缩略图
127  bitmap.Save(thumbnailPath, System.Drawing.Imaging.ImageFormat.Jpeg);
128  }
129  finally
130  {
131  originalImage.Dispose();
132  bitmap.Dispose();
133  g.Dispose();
134  }
135  }
136 
137  #endregion 生成缩略图
138  }
