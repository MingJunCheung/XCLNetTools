723  public static string GetQuerySerializeString()
724  {
725  var req = HttpContext.Current.Request;
726  if (null == req.QueryString || req.QueryString.Count == 0)
727  {
728  return string.Empty;
729  }
730  List<string> lst = new List<string>();
731  string[] vals = new string[] { };
732  for (int i = 0; i < req.QueryString.Count; i++)
733  {
734  vals = req.QueryString.GetValues(i);
735  for (int j = 0; j < vals.Length; j++)
736  {
737  lst.Add(string.Format("{0}={1}", req.QueryString.GetKey(i), vals[j]));
738  }
739  }
740  return string.Join("&", lst.ToArray());
741  }
