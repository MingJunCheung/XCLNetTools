727  public static string GetQuerySerializeString()
728  {
729  var req = HttpContext.Current.Request;
730  if (null == req.QueryString || req.QueryString.Count == 0)
731  {
732  return string.Empty;
733  }
734  List<string> lst = new List<string>();
735  string[] vals = new string[] { };
736  for (int i = 0; i < req.QueryString.Count; i++)
737  {
738  vals = req.QueryString.GetValues(i);
739  for (int j = 0; j < vals.Length; j++)
740  {
741  lst.Add(string.Format("{0}={1}", req.QueryString.GetKey(i), vals[j]));
742  }
743  }
744  return string.Join("&", lst.ToArray());
745  }
